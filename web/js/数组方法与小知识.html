      # 扁平化数组：flat()

      Array.prototype.flat() (opens new window)（预计将成为 ES2019 的一部分，并且已在主要浏览器中实现），如不支持，参考polyfill补丁代码：

      var arr = [1,[2,3],[4,[5,6]]]
function flatten(arr) {
  const result = [];

  arr.forEach((i) => {
    if (Array.isArray(i))
      result.push(...flatten(i));
    else
      result.push(i);
  })
  
  return result;
}
flatten(arr)//[1, 2, 3, 4, 5, 6]
